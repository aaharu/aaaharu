package urulab.fp10.core 
{
    import away3d.containers.ObjectContainer3D;
    import away3d.events.Loader3DEvent;
    import away3d.loaders.Collada;
    import away3d.loaders.Loader3D;
    import away3d.materials.PhongBitmapMaterial;
    
    import flash.display.Loader;
    import flash.events.Event;
    import flash.events.ProgressEvent;
    import flash.net.URLRequest;
    import flash.text.TextField;
    
    /**
     * 
     * UruLabViewクラスは「うるらぼ」の閲覧用クラスです。
     * 
     * @author otake haruaki
     */
    public class UruLabView extends UruLab3D
    {
        // モデルオブジェクト
        private var model:ObjectContainer3D;
        // モデルのマテリアル
        private var material:PhongBitmapMaterial;
        // ロードプログレスの表示
        private var tf:TextField;
        
        /**
         * constructor
         */
        public function UruLabView(w:int, h:int, colladaURI:String, textureURI:String)
        {
            super(w, h);
            // テクスチャのロード
            var loader:Loader = new Loader();
            loader.contentLoaderInfo.addEventListener(ProgressEvent.PROGRESS, loadMaterialProgress);
            loader.contentLoaderInfo.addEventListener(Event.COMPLETE, loadMaterialComplete(colladaURI));
            loader.load(new URLRequest(textureURI));
            tf = new TextField();
            tf.x = tf.y = 250;
            addChild(tf);
        }
        
        /**
         * テクスチャ画像ロードのプログレスイベントリスナー
         */
        protected function loadMaterialProgress(e:ProgressEvent):void
        {
            tf.text = "ロード中\n" + (int(50 * e.bytesLoaded / e.bytesTotal)).toString() + " / 100";
        }
        /**
         * テクスチャ画像のロード完了
         */
        protected function loadMaterialComplete(colladaURI:String):Function
        {
            return function(e:Event):void
            {
                material = new PhongBitmapMaterial(e.target.content.bitmapData);
                // モデルのロード
                var loader:Loader3D = Collada.load(colladaURI);
                loader.addEventListener(Loader3DEvent.LOAD_PROGRESS, loadColladaProgress);
                loader.addEventListener(Loader3DEvent.LOAD_SUCCESS, loadColladaComplete);
            };
        }
        
        /**
         * モデルのロードプログレスイベントリスナー
         */
        protected function loadColladaProgress(e:Loader3DEvent):void
        {
            tf.text = "ロード中\n" + (50 + int(50 * e.loader.bytesLoaded / e.loader.bytesTotal)).toString() + " / 100";
        }
        
        /**
         * モデルのロード完了
         */
        protected function loadColladaComplete(e:Loader3DEvent):void
        {
            removeChild(tf);
            model = e.loader.handle as ObjectContainer3D;
            model.materialLibrary.getMaterial("UV_map").material = material;
            scene.addChild(model);
        }
    }
}
