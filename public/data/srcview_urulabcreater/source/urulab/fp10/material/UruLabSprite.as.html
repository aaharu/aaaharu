package urulab.fp10.material
{
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Graphics;
    import flash.display.Sprite;
    import flash.geom.Point;
    import flash.geom.Rectangle;
    
    import urulab.fp10.core.UruLab3D;
    
    public class UruLabSprite extends Sprite
    {
        [Embed("../../../../../UruLabCreater/src/image/rankaku1.png")]private var rankaku:Class;
        private var _base:Bitmap;
        private var _middle:Bitmap;
        private var _surface:Sprite;
        private var _g:Graphics;
        private var _figure:Sprite;
        
        private var speed:Point;
        private var position:Point;
        private var thickness:Number = 1;
        
        public function get base():Bitmap
        {
            return _base;
        }
        
        public function get bitmapData():BitmapData
        {
            return _base.bitmapData;
        }
        
        public function get middle():Bitmap
        {
            return _middle;
        }
        
        public function get surface():Sprite
        {
            return _surface;
        }
        
        public function UruLabSprite(bmp:Bitmap)
        {
            // spは下base(Bitmap)、上surface(Sprite)の2層構造　じゃなくなりました
            _base = bmp;
            _middle = new Bitmap(new BitmapData(UruLab3D.TEX_WIDTH, UruLab3D.TEX_HEIGHT, true, 0xffffffff), "auto", true);
            _surface = new Sprite();
            _g = _surface.graphics;
            this.addChild(_base);
            this.addChild(_surface);
            super();
        }
        
        public function penMoveTo(size:int, color:uint, u:Number, v:Number):void
        {
            _g.lineStyle(size, color);
            _g.moveTo(u * UruLab3D.TEX_WIDTH, (1 - v) * UruLab3D.TEX_HEIGHT);
        }
        
        public function pen2MoveTo(size:int, color:uint, u:Number, v:Number):void
        {
            speed = new Point();
            position = new Point(u * UruLab3D.TEX_WIDTH, (1 - v) * UruLab3D.TEX_HEIGHT);
            _g.lineStyle(size / 3, color);
            _g.moveTo(u * UruLab3D.TEX_WIDTH, (1 - v) * UruLab3D.TEX_HEIGHT);
        }
        
        public function pen2LineTo(size:int, color:uint, u:Number, v:Number):void
        {
            for(var i:int = 1; i <= 7; ++i)
            {
                speed.x += ((position.x + i / 7 * (u * UruLab3D.TEX_WIDTH - position.x)) - position.x) / 30;
                speed.y += ((position.y + i / 7 * ((1 - v) * UruLab3D.TEX_HEIGHT - position.y)) - position.y) / 30;
                speed.x /= 1.2;
                speed.y /= 1.2;
                position.x += speed.x;
                position.y += speed.y;
                
                var s:Number = Math.sqrt(speed.x * speed.x + speed.y * speed.y);
                thickness += (s * 2 - thickness) / 8.5;
                thickness = Math.min(Math.max(thickness, 1.5), 10);
                
                _g.lineStyle(thickness * size / 3, color);
                _g.lineTo(position.x, position.y);
            }
        }
        
        public function pen2Draw():void
        {
            drawSurface();
            thickness = 1;
        }
        
        public function showBorder(x:Number, y:Number, w:Number, h:Number):void
        {
            // モデル上にマウスカーソルがあるときカーソル周りにスタンプの大きさの枠を表示
            _g.clear();
            _g.lineStyle(1, 0xffffff, 0.5);
            _g.drawRect(x, y, w, h);
        }
        
        public function rankakuDown(x:Number, y:Number, size:Number):void
        {
            _base.bitmapData.draw(new rankaku() as Bitmap, null, null, null, new Rectangle(x, y, size, size));
        }
        
        public function drawCircle(x:Number, y:Number, r:Number, color:uint):void
        {
            _g.beginFill(color);
            _g.drawCircle(x, y, r);
            _g.endFill();
        }
        
        public function drawSurface():void
        {
            _base.bitmapData.draw(_surface, null, null, null, null, true);
            // surfaceに残っている線を消す
            _g.clear();
        }
    }
}
