package
{
    import flash.display.BitmapData;
    import flash.display.Sprite;

    /**
    * BindableのObjectを作るクラス
    */
    public class SetData
    {
        [Bindable]public var name:String;
        [Bindable]public var creater:String;
        [Bindable]public var bmpd:BitmapData;
        [Bindable]public var sp:Sprite;
        
        public function SetData(name:String, creater:String, bmpd:BitmapData, sp:Sprite = null)
        {
            this.name = name;
            this.creater = creater;
            this.bmpd = bmpd;
            this.sp = sp;
        }
    }
}
