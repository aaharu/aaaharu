package urulab.fp10.material
{
    import away3d.materials.BitmapMaterial;
    import away3d.materials.MovieMaterial;
    import away3d.materials.PhongMovieMaterial;

    public class UruLabMaterial
    {
        private var _shaded:Boolean = true;
        private var onLite:PhongMovieMaterial;
        private var offLite:MovieMaterial;
        
        public function set shade(bool:Boolean):void
        {
            _shaded = bool;
        }
        
        public function get material():BitmapMaterial
        {
            if(_shaded) {
                return onLite;
            } else {
                return offLite;
            }
        }
        
        public function UruLabMaterial(sp:UruLabSprite)
        {
            // ライトありとライトなし用のマテリアル
            onLite = new PhongMovieMaterial(sp);
            offLite = new MovieMaterial(sp);
            
            offLite.smooth = true;
            //offLite.interactive = true;
            offLite.debug = true;
            offLite.wireColor = 0xffffff;
        }
    }
}
