package urulab.fp10.core 
{
    import away3d.containers.ObjectContainer3D;
    import away3d.events.Loader3DEvent;
    import away3d.events.MouseEvent3D;
    import away3d.loaders.Collada;
    import away3d.loaders.Loader3D;
    
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Loader;
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.events.ProgressEvent;
    import flash.geom.ColorTransform;
    import flash.geom.Matrix;
    import flash.geom.Point;
    import flash.geom.Rectangle;
    import flash.net.URLRequest;
    import flash.text.TextField;
    
    import urulab.fp10.material.UruLabMaterial;
    import urulab.fp10.material.UruLabSprite;
    
    /**
     * 
     * UruLabEditクラスは「うるらぼ」の編集用クラスです。
     * 
     * @author otake haruaki
     */
    public class UruLabEdit extends UruLab3D
    {
        // よく使うからまとめた
        private const TEX_RECT:Rectangle = new Rectangle(0, 0, TEX_WIDTH, TEX_HEIGHT);
        private const POINT_ZERO:Point = new Point();
        // モデルオブジェクト
        private var model:ObjectContainer3D;
        // モデルのマテリアル
        private var material:UruLabMaterial;
        // マテリアル用スプライト
        private var _sprite:UruLabSprite;
        
        // ビュー、スタンプ、筆、消しゴム
        private var _mode:String;
        // 消しゴム用の色
        private var _color:uint;
        
        // svg初期幅
        private var svgWidth:Number = 0;
        // 「元に戻す」用Vector
        private var undoData:Vector.<BitmapData> = new Vector.<BitmapData>();
        // 「やり直し」用Vector
        private var redoData:Vector.<BitmapData> = new Vector.<BitmapData>();
        // スタンプのビットマップデータ
        private var stampBmpd:BitmapData;
        // スタンプのSVGデータ
        private var stampSVG:Sprite;
        // ペンの大きさ
        private var _penSize:int = 5;
        // 消しゴムの大きさ
        private var _eraserSize:int = 5;
        // ペンの色
        private var _penColor:uint = 0xffffff;
        // ロードプログレスの表示
        private var tf:TextField;
        
        /**
         * 現在のモード取得
         * @return String
         */
        public function get mode():String
        {
            return _mode;
        }
        
        /**
         * モードのセット
         * @param name モード名
         */
        public function set mode(name:String):void
        {
            _mode = name;
            if(_mode == UruLabMode.VIEW)
                runViewMode();
            else
                runEditMode();
        }
        
        /**
         * 
         * @return BitmapData
         */
        public function get texture():BitmapData
        {
            var bmpd:BitmapData = new BitmapData(TEX_WIDTH, TEX_HEIGHT);
            bmpd.copyPixels(_sprite.bitmapData, TEX_RECT, POINT_ZERO);
            return bmpd;
        }
        
        /**
         * ライトが点いているかどうか
         * @param bool ライト点灯フラグ
         */
        public function set lighting(bool:Boolean):void
        {
            if(model)
            {
                material.shade = bool;
                model.materialLibrary.getMaterial("UV_map").material = material.material;
            }
        }
        
        /**
         * スタンプ
         * @param obj スタンプ情報連想配列
         */
        public function set stamp(obj:Object):void
        {
            if(obj.bmp)
                stampBmpd = obj.bmp;
            else
            {
                stampSVG = obj.svg;
                svgWidth = obj.svg.width;
            }
        }
        
        /**
         * ペンの大きさセット
         * @param value 太さの値
         */
        public function set penSize(value:int):void
        {
            _penSize = value;
        }
        
        /**
         * 消しゴムの大きさセット
         * @param value 太さの値
         */
        public function set eraserSize(value:int):void
        {
            _eraserSize = value;
        }
        
        /**
         * ペンの色セット
         * @param color 色
         */
        public function set penColor(color:uint):void
        {
            _penColor = color;
        }
        
        /**
         * constructor
         */
        public function UruLabEdit(w:int, h:int, colladaURI:String, color:uint, textureURI:String = null)
        {
            super(w, h);
            // ARGB to RGB
            _color = color - 0xFF000000;
            if(textureURI)
            {
                // テクスチャがあればテクスチャのロード
                var loader:Loader = new Loader();
                loader.contentLoaderInfo.addEventListener(ProgressEvent.PROGRESS, loadMaterialProgress);
                loader.contentLoaderInfo.addEventListener(Event.COMPLETE, loadMaterialComplete(colladaURI));
                loader.load(new URLRequest(textureURI));
            }
            else
            {
                // テクスチャがない場合は単色
                _sprite = new UruLabSprite(new Bitmap(new BitmapData(TEX_WIDTH, TEX_HEIGHT, true, color), "auto", true));
                // モデルのロード
                var loader3d:Loader3D = Collada.load(colladaURI);
                loader3d.addEventListener(Loader3DEvent.LOAD_PROGRESS, loadColladaProgress);
                loader3d.addEventListener(Loader3DEvent.LOAD_SUCCESS, loadColladaComplete);
            }
            tf = new TextField();
            tf.x = tf.y = 250;
            addChild(tf);
        }
        
        /**
         * テクスチャ画像ロードのプログレスイベント
         * @param e
         */
        private function loadMaterialProgress(e:ProgressEvent):void
        {
            tf.text = "ロード中\n" + (int(50 * e.bytesLoaded / e.bytesTotal)).toString() + " / 100";
        }
        /**
         * テクスチャ画像のロード完了
         * @param colladaURI モデルデータ(Collada形式)のURL
         */
        private function loadMaterialComplete(colladaURI:String):Function
        {
            return function(e:Event):void
            {
                _sprite = new UruLabSprite(new Bitmap(e.target.content.bitmapData, "auto", true));
                // モデルのロード
                var loader3d:Loader3D = Collada.load(colladaURI);
                loader3d.addEventListener(Loader3DEvent.LOAD_PROGRESS, loadColladaProgress);
                loader3d.addEventListener(Loader3DEvent.LOAD_SUCCESS, loadColladaComplete);
            };
        }
        
        /**
         * モデルのロードプログレスイベント
         * @param e
         */
        private function loadColladaProgress(e:Loader3DEvent):void
        {
            tf.text = "ロード中\n" + (50 + int(50 * e.loader.bytesLoaded / e.loader.bytesTotal)).toString() + " / 100";
        }
        
        /**
         * モデルのロード完了
         * @param e
         */
        private function loadColladaComplete(e:Loader3DEvent):void
        {
            removeChild(tf);
            model = e.loader.handle as ObjectContainer3D;
            
            material = new UruLabMaterial(_sprite);
            
            // 最初はライト点灯
            lighting = true;
            mode = UruLabMode.VIEW;
            scene.addChild(model);
        }
        
        /**
         * ビューモードに切り替え
         */
        private function runViewMode():void
        {
            if(!parent.parent.parent.parent.hasEventListener(MouseEvent.MOUSE_DOWN))
            {
                // カメラ移動のイベントリスナー追加
                parent.parent.parent.parent.addEventListener(MouseEvent.MOUSE_DOWN, onMouseDown);
                parent.parent.parent.parent.addEventListener(MouseEvent.MOUSE_UP, onMouseUp);
            }
            if(model)
            {
                // 編集用のイベントリスナー削除
                model.removeOnMouseOver(onModelMouseOverBMP);
                model.removeOnMouseDown(onModelMouseDownSVG);
                model.removeOnMouseDown(onModelMouseDownPen);
                model.removeOnMouseDown(onModelMouseDownPen2);
                model.removeOnMouseDown(onModelMouseDownMatPen);
                model.removeOnMouseDown(onModelMouseDownEraser);
            }
        }
        
        /**
         * 編集モードに切り替え
         */
        private function runEditMode():void
        {
            if(parent.parent.parent.parent.hasEventListener(MouseEvent.MOUSE_DOWN))
            {
                // カメラ移動のイベントリスナー削除 parent.parent.parent.parentでmodelViewer
                parent.parent.parent.parent.removeEventListener(MouseEvent.MOUSE_DOWN, onMouseDown);
                parent.parent.parent.parent.removeEventListener(MouseEvent.MOUSE_UP, onMouseUp);
            }
            if(model)
            {
                // modeによってイベント分けてます　無理やり
                switch(mode)
                {
                    case UruLabMode.STAMP_BMP:
                        model.addOnMouseOver(onModelMouseOverBMP);
                        model.removeOnMouseDown(onModelMouseDownSVG);
                        model.removeOnMouseDown(onModelMouseDownPen);
                        model.removeOnMouseDown(onModelMouseDownPen2);
                        model.removeOnMouseDown(onModelMouseDownMatPen);
                        model.removeOnMouseDown(onModelMouseDownEraser);
                        break;
                    case UruLabMode.STAMP_SVG:
                        model.removeOnMouseOver(onModelMouseOverBMP);
                        model.addOnMouseDown(onModelMouseDownSVG);
                        model.removeOnMouseDown(onModelMouseDownPen);
                        model.removeOnMouseDown(onModelMouseDownPen2);
                        model.removeOnMouseDown(onModelMouseDownMatPen);
                        model.removeOnMouseDown(onModelMouseDownEraser);
                        break;
                    case UruLabMode.PEN:
                        model.removeOnMouseOver(onModelMouseOverBMP);
                        model.removeOnMouseDown(onModelMouseDownSVG);
                        model.addOnMouseDown(onModelMouseDownPen);
                        model.removeOnMouseDown(onModelMouseDownPen2);
                        model.removeOnMouseDown(onModelMouseDownMatPen);
                        model.removeOnMouseDown(onModelMouseDownEraser);
                        break;
                    case UruLabMode.PEN2:
                        model.removeOnMouseOver(onModelMouseOverBMP);
                        model.removeOnMouseDown(onModelMouseDownSVG);
                        model.removeOnMouseDown(onModelMouseDownPen);
                        model.addOnMouseDown(onModelMouseDownPen2);
                        model.removeOnMouseDown(onModelMouseDownMatPen);
                        model.removeOnMouseDown(onModelMouseDownEraser);
                        break;
                    case UruLabMode.MAT_PEN:
                        model.removeOnMouseOver(onModelMouseOverBMP);
                        model.removeOnMouseDown(onModelMouseDownSVG);
                        model.removeOnMouseDown(onModelMouseDownPen);
                        model.removeOnMouseDown(onModelMouseDownPen2);
                        model.addOnMouseDown(onModelMouseDownMatPen);
                        model.removeOnMouseDown(onModelMouseDownEraser);
                        break;
                    case UruLabMode.ERASER:
                        model.removeOnMouseOver(onModelMouseOverBMP);
                        model.removeOnMouseDown(onModelMouseDownSVG);
                        model.removeOnMouseDown(onModelMouseDownPen);
                        model.removeOnMouseDown(onModelMouseDownPen2);
                        model.removeOnMouseDown(onModelMouseDownMatPen);
                        model.addOnMouseDown(onModelMouseDownEraser);
                        break;
                    default:
                        model.removeOnMouseOver(onModelMouseOverBMP);
                        model.removeOnMouseDown(onModelMouseDownSVG);
                        model.removeOnMouseDown(onModelMouseDownPen);
                        model.removeOnMouseDown(onModelMouseDownPen2);
                        model.removeOnMouseDown(onModelMouseDownMatPen);
                        model.removeOnMouseDown(onModelMouseDownEraser);
                        break;
                }
            }
        }
        
        private function onModelMouseDownPen(e:MouseEvent3D):void
        {
            // マウスダウンしたら他のイベントを追加
            model.addOnMouseUp(onModelMouseUpPen);
            model.addOnMouseOut(onModelMouseUpPen);
            model.addOnMouseMove(onModelMouseMovePen);
            
            pushUndoData();
            // surfaceに描画
            _sprite.penMoveTo(_penSize, _penColor, e.uv.u, e.uv.v);
        }
        private function onModelMouseMovePen(e:MouseEvent3D):void
        {
            _sprite.surface.graphics.lineTo(e.uv.u * TEX_WIDTH, (1 - e.uv.v) * TEX_HEIGHT);
        }
        private function onModelMouseUpPen(e:MouseEvent3D):void
        {
            _sprite.drawSurface();
            
            // マウスアップしたら一連のイベントを消去
            model.removeOnMouseUp(onModelMouseUpPen);
            model.removeOnMouseOut(onModelMouseUpPen);
            model.removeOnMouseMove(onModelMouseMovePen);
        }
        
        private function onModelMouseDownMatPen(e:MouseEvent3D):void
        {
            // マウスダウンしたら他のイベントを追加
            model.addOnMouseUp(onModelMouseUpMatPen);
            model.addOnMouseOut(onModelMouseUpMatPen);
            model.addOnMouseMove(onModelMouseMoveMatPen);
            
            pushUndoData();
            
            // BMPはbaseに直接draw　アルファ情報もってるのでcopyPixelsではなくdrawを使う
            var h:int = _penSize >> 1;
            _sprite.rankakuDown(e.uv.u * TEX_WIDTH - h, (1 - e.uv.v) * TEX_HEIGHT - h, _penSize);
        }
        private function onModelMouseMoveMatPen(e:MouseEvent3D):void
        {
            var h:int = _penSize >> 1;
            _sprite.rankakuDown(e.uv.u * TEX_WIDTH - h, (1 - e.uv.v) * TEX_HEIGHT - h, _penSize);
        }
        private function onModelMouseUpMatPen(e:MouseEvent3D):void
        {
            // マウスアップしたら一連のイベントを消去
            model.removeOnMouseUp(onModelMouseUpMatPen);
            model.removeOnMouseOut(onModelMouseUpMatPen);
            model.removeOnMouseMove(onModelMouseMoveMatPen);
        }
        
        /*
        onModelMouseUpPen2, onModelMouseUpPen2, onModelMouseMovePen2からによるPen2の制作において、
        jozefchutka氏による「Penice（http://wonderfl.net/c/dbrQ）」を参考にしています。
        UruLabに適用させるため、上記クラスを改変しています。
        
        Thanks, jozefchutka!
        */
        private function onModelMouseDownPen2(e:MouseEvent3D):void
        {
            // マウスダウンしたら他のイベントを追加
            model.addOnMouseUp(onModelMouseUpPen2);
            model.addOnMouseOut(onModelMouseUpPen2);
            model.addOnMouseMove(onModelMouseMovePen2);
            
            pushUndoData();
            _sprite.pen2MoveTo(_penSize, _penColor, e.uv.u, e.uv.v);
        }
        private function onModelMouseMovePen2(e:MouseEvent3D):void
        {
            _sprite.pen2LineTo(_penSize, _penColor, e.uv.u, e.uv.v);
        }
        private function onModelMouseUpPen2(e:MouseEvent3D):void
        {
            _sprite.pen2Draw();
            
            // マウスアップしたら一連のイベントを消去
            model.removeOnMouseUp(onModelMouseUpPen2);
            model.removeOnMouseOut(onModelMouseUpPen2);
            model.removeOnMouseMove(onModelMouseMovePen2);
        }
        
        private function onModelMouseOverBMP(e:MouseEvent3D):void
        {
            // BMPモードではモデル上にマウスカーソルが重なったら必要なイベントを追加
            model.addOnMouseMove(onModelMouseMoveBMP);
            model.addOnMouseDown(onModelMouseDownBMP);
            model.addOnMouseOut(onModelMouseOutBMP);
        }
        private function onModelMouseMoveBMP(e:MouseEvent3D):void
        {
            _sprite.showBorder(e.uv.u * TEX_WIDTH - (stampBmpd.width >> 1), (1 - e.uv.v) * TEX_HEIGHT - (stampBmpd.height >> 1), stampBmpd.width, stampBmpd.height);
        }
        private function onModelMouseDownBMP(e:MouseEvent3D):void
        {
            pushUndoData();
            
            // BMPスタンプはbaseに直接draw　アルファ情報もってるのでcopyPixelsではなくdrawを使う
            var matrix:Matrix = new Matrix();
            matrix.translate(e.uv.u * TEX_WIDTH - (stampBmpd.width >> 1), (1 - e.uv.v) * TEX_HEIGHT - (stampBmpd.height >> 1));
            _sprite.bitmapData.draw(stampBmpd, matrix, new ColorTransform(0, 0, 0, 1, _penColor >> 16 & 0xFF, _penColor >> 8  & 0xFF, _penColor & 0xFF));
        }
        private function onModelMouseOutBMP(e:MouseEvent3D):void
        {
            // モデル上にマウスカーソルがない場合は枠を消去
            _sprite.surface.graphics.clear();
            
            model.removeOnMouseMove(onModelMouseMoveBMP);
            model.removeOnMouseDown(onModelMouseDownBMP);
            model.removeOnMouseOut(onModelMouseOutBMP);
        }
        
        private function onModelMouseDownSVG(e:MouseEvent3D):void
        {
            // マウスダウンしたら他のイベントを追加
            model.addOnMouseUp(onModelMouseUpSVG);
            model.addOnMouseOut(onModelMouseUpSVG);
            model.addOnMouseMove(onModelMouseMoveSVG);
            
            pushUndoData();
            stampSVG.x = e.uv.u * TEX_WIDTH;
            stampSVG.y = (1 - e.uv.v) * TEX_HEIGHT;
            _sprite.surface.addChild(stampSVG);
        }
        private function onModelMouseMoveSVG(e:MouseEvent3D):void
        {
            // モデル上でのマウスの移動距離
            var dx:Number = e.uv.u * TEX_WIDTH - stampSVG.x;
            var dy:Number = (1 - e.uv.v) * TEX_HEIGHT - stampSVG.y;
            // 移動距離による大きさ計算
            var scale:Number = (Math.sqrt(dx * dx + dy * dy) / svgWidth) * 2;
            var matrix:Matrix = new Matrix();
            // クリックした点から動いた角度回転
            matrix.rotate(Math.atan2(dy, dx));
            matrix.scale(scale, scale);
            matrix.translate(stampSVG.x, stampSVG.y);
            stampSVG.transform.matrix = matrix;
        }
        private function onModelMouseUpSVG(e:MouseEvent3D):void
        {
            // surfaceをbaseにdraw
            _sprite.bitmapData.draw(_sprite.surface, null, null, null, null, true);
            // surfaceに残ってるものを削除
            _sprite.surface.removeChild(stampSVG);
            
            // マウスアップかマウスアウトをしたら一連のイベントを消去
            model.removeOnMouseUp(onModelMouseUpSVG);
            model.removeOnMouseOut(onModelMouseUpSVG);
            model.removeOnMouseMove(onModelMouseMoveSVG);
        }
        
        private function onModelMouseDownEraser(e:MouseEvent3D):void
        {
            model.addOnMouseUp(onModelMouseUpEraser);
            model.addOnMouseOut(onModelMouseUpEraser);
            model.addOnMouseMove(onModelMouseMoveEraser);
            
            pushUndoData();
            // surfaceに描画
            _sprite.drawCircle(e.uv.u * TEX_WIDTH, (1 - e.uv.v) * TEX_HEIGHT, _eraserSize, _color);
        }
        private function onModelMouseMoveEraser(e:MouseEvent3D):void
        {
            // 線（円）を描画
            _sprite.drawCircle(e.uv.u * TEX_WIDTH, (1 - e.uv.v) * TEX_HEIGHT, _eraserSize, _color);
        }
        private function onModelMouseUpEraser(e:MouseEvent3D):void
        {
            // surfaceをbaseにdraw
            _sprite.drawSurface();
            
            model.removeOnMouseUp(onModelMouseUpEraser);
            model.removeOnMouseOut(onModelMouseUpEraser);
            model.removeOnMouseMove(onModelMouseMoveEraser);
        }
        
        /**
         * UndoDataに現在のbaseをpush
         */
        private function pushUndoData():void
        {
            // 描く前の状態をVectorに追加
            var bmpData:BitmapData = new BitmapData(TEX_WIDTH, TEX_HEIGHT);
            bmpData.copyPixels(_sprite.bitmapData, TEX_RECT, POINT_ZERO);
            if(undoData.length > 30)
                undoData.shift();
            undoData.push(bmpData);
        }
        
        /**
         * 元に戻す
         */
        public function undo():void
        {
            if(undoData.length > 0)
            {
                var bmpData:BitmapData = new BitmapData(TEX_WIDTH, TEX_HEIGHT);
                bmpData.copyPixels(_sprite.bitmapData, TEX_RECT, POINT_ZERO);
                if(redoData.length > 30)
                    redoData.shift();
                redoData.push(bmpData);
                
                var bmpd:BitmapData = undoData.pop();
                _sprite.bitmapData.copyPixels(bmpd, TEX_RECT, POINT_ZERO);
            }
        }
        
        /**
         * やり直し
         */
        public function redo():void
        {
            if(redoData.length > 0)
            {
                var bmpData:BitmapData = new BitmapData(TEX_WIDTH, TEX_HEIGHT);
                bmpData.copyPixels(_sprite.bitmapData, TEX_RECT, POINT_ZERO);
                undoData.push(bmpData);
                
                var bmpd:BitmapData = redoData.pop();
                _sprite.bitmapData.copyPixels(bmpd, TEX_RECT, POINT_ZERO);
            }
        }
    }
}
