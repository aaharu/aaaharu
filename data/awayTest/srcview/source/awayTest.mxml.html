<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" creationComplete="init()" viewSourceURL="srcview/index.html">
    <mx:Canvas id="myCanvas" x="0" y="0" width="400" height="400" backgroundColor="0xFFFFFF" />
    <mx:HRule x="408" y="56" width="160" />
    <mx:Label x="408" y="10" text="ambient" />
    <mx:HSlider id="amb" x="408" y="36" minimum="0" maximum="1.0" value="0.25" change="change(event)" />
    <mx:HRule x="408" y="10" width="160" />
    <mx:Label x="408" y="66" text="diffuse" />
    <mx:HSlider id="dif" x="408" y="90" minimum="0" maximum="1.0" value="0.75" change="change(event)" />
    <mx:HRule x="408" y="110" width="160" />
    <mx:Label x="408" y="120" text="specular" />
    <mx:HSlider id="spe" x="408" y="146" minimum="0" maximum="1.0" value="0.9" change="change(event)" />
    <mx:HRule x="408" y="166" width="160" />
    <mx:Script>
        <![CDATA[
            import mx.controls.SWFLoader;
            private var world:World;

            private function init():void {
                world = new World(myCanvas);
                var swfLoader:SWFLoader = new SWFLoader();
                swfLoader.source = world;
                myCanvas.addChild(swfLoader);
            }
            private function change(e:Event):void {
                switch(e.target.id) {
                    case "amb":
                        world.changeAmbient(e.target.value);
                        break;
                    case "dif":
                        world.changeDiffuse(e.target.value);
                        break;
                    case "spe":
                        world.changeSpecular(e.target.value);
                        break;
                    default:
                        break;
                }
            }
        ]]>
    </mx:Script>
</mx:Application>
