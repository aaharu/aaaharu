package awaybuilder.utils
{
    public class CoordinateCopy
    {
        public static const X : String = "x" ;
        public static const Y : String = "y" ;
        public static const Z : String = "z" ;
        public static const ROTATION_X : String = "rotationX" ;
        public static const ROTATION_Y : String = "rotationY" ;
        public static const ROTATION_Z : String = "rotationZ" ;
        public static const SCALE_X : String = "scaleX" ;
        public static const SCALE_Y : String = "scaleY" ;
        public static const SCALE_Z : String = "scaleZ" ;
        
        
        
        public static function position ( source : * , target : * ) : void
        {
            target[ CoordinateCopy.X ] = source[ CoordinateCopy.X ] ;
            target[ CoordinateCopy.Y ] = source[ CoordinateCopy.Y ] ;
            target[ CoordinateCopy.Z ] = source[ CoordinateCopy.Z ] ;
        }
        
        
        
        public static function rotation ( source : * , target : * ) : void
        {
            target[ CoordinateCopy.ROTATION_X ] = source[ CoordinateCopy.ROTATION_X ] ;
            target[ CoordinateCopy.ROTATION_Y ] = source[ CoordinateCopy.ROTATION_Y ] ;
            target[ CoordinateCopy.ROTATION_Z ] = source[ CoordinateCopy.ROTATION_Z ] ;
        }
        
        
        
        public static function scale ( source : * , target : * ) : void
        {
            target[ CoordinateCopy.SCALE_X ] = source[ CoordinateCopy.SCALE_X ] ;
            target[ CoordinateCopy.SCALE_Y ] = source[ CoordinateCopy.SCALE_Y ] ;
            target[ CoordinateCopy.SCALE_Z ] = source[ CoordinateCopy.SCALE_Z ] ;
        }
    }
}
